💡 Google Apps Script 백엔드 설정 가이드

글 등록·수정·삭제·조회수 자동 증가가 모두 동작하려면, 구글 시트에 아래 스크립트를 배포해야 합니다. 기존에 생성(Create)만 지원하던 스크립트를 사용 중이라면, 아래 코드로 교체 배포해주세요.

전체 Apps Script 코드 보기 (doGet + doPost)
[세팅 방법]
1. 연동할 구글 스프레드시트 열기 → 상단 메뉴 [확장 프로그램] → [Apps Script]
2. 기존 코드를 지우고 아래 코드를 붙여넣기
3. 시트 1행(헤더)을 id, category, emoji, title, summary, tags, date, content, views 순서로 맞추기
4. 우측 상단 [배포] → [새 배포] (유형: 웹 앱, 실행: 나, 액세스 권한: 모든 사용자) 후 URL 발급
5. 발급받은 URL을 assets/site.js 파일의 WEB_APP_URL 값에 붙여넣기 (index.html, admin.html 전체에 자동 반영)
function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; var data = sheet.getDataRange().getValues(); var headers = data[0]; var rows = data.slice(1).map(function(row) { var obj = {}; headers.forEach(function(h, i) { obj[h] = row[i]; }); return obj; }); return ContentService.createTextOutput(JSON.stringify({ result: "success", posts: rows })) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; var params = JSON.parse(e.postData.contents); var action = params.action || "create"; var data = sheet.getDataRange().getValues(); var headers = data[0]; var idCol = headers.indexOf("id"); function findRowIndexById(id) { for (var i = 1; i < data.length; i++) { if (String(data[i][idCol]) === String(id)) return i + 1; // 시트 행번호(1-base) } return -1; } if (action === "create") { sheet.appendRow([ params.id, params.category, params.emoji, params.title, params.summary, params.tags, params.date, params.content, params.views || 0 ]); } else if (action === "update") { var rowNum = findRowIndexById(params.id); if (rowNum === -1) { return ContentService.createTextOutput(JSON.stringify({ result: "error", message: "ID를 찾을 수 없습니다." })) .setMimeType(ContentService.MimeType.JSON); } var newRow = [ params.id, params.category, params.emoji, params.title, params.summary, params.tags, params.date, params.content, params.views != null ? params.views : data[rowNum - 1][headers.indexOf("views")] ]; sheet.getRange(rowNum, 1, 1, newRow.length).setValues([newRow]); } else if (action === "delete") { var rowNum = findRowIndexById(params.id); if (rowNum === -1) { return ContentService.createTextOutput(JSON.stringify({ result: "error", message: "ID를 찾을 수 없습니다." })) .setMimeType(ContentService.MimeType.JSON); } sheet.deleteRow(rowNum); } else if (action === "incrementView") { var rowNum = findRowIndexById(params.id); if (rowNum !== -1) { var viewsCol = headers.indexOf("views") + 1; var current = Number(sheet.getRange(rowNum, viewsCol).getValue()) || 0; sheet.getRange(rowNum, viewsCol).setValue(current + 1); } } return ContentService.createTextOutput(JSON.stringify({ result: "success" })) .setMimeType(ContentService.MimeType.JSON); }
