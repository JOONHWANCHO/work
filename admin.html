💡 Google Apps Script 백엔드 설정 가이드

글 등록·수정·삭제·조회수 자동 증가가 모두 동작하려면, 구글 시트에 아래 스크립트를 배포해야 합니다. 기존에 생성(Create)만 지원하던 스크립트를 사용 중이라면, 아래 코드로 교체 배포해주세요.

전체 Apps Script 코드 보기 (doGet + doPost + 분석로그)
[세팅 방법]
1. 연동할 구글 스프레드시트 열기 → 상단 메뉴 [확장 프로그램] → [Apps Script]
2. 기존 코드를 지우고 아래 코드를 붙여넣기
3. 게시글 시트(예: 시트1) 1행 헤더를 id, category, emoji, title, summary, tags, date, content, views 순서로 맞추기
4. 분석로그 시트는 코드가 처음 실행될 때 "분석로그"라는 이름으로 자동 생성되며 헤더도 자동으로 채워집니다 (직접 만들 필요 없음)
5. 우측 상단 [배포] → [새 배포] (유형: 웹 앱, 실행: 나, 액세스 권한: 모든 사용자) 후 URL 발급
6. 발급받은 URL을 assets/site.js 파일의 WEB_APP_URL 값에 붙여넣기 (전체 사이트에 자동 반영)
// ============================================================ // 모두의 계산기 — Apps Script 백엔드 // 게시글 CRUD + 방문자/트래픽/수익 분석 로그 수집 // ============================================================ var POSTS_SHEET = "시트1"; // 게시글 시트 이름 var LOG_SHEET = "분석로그"; // 분석 로그 시트 이름 (자동 생성됨) var LOG_HEADERS = [ "timestamp","event_type","session_id","visitor_id","is_new_visitor", "page_path","page_title","referrer","traffic_source","utm_medium","utm_campaign", "device_type","browser","os","country","entry_page", "duration_sec","exit_type","target_url","link_category","calculator_name" ]; function getPostsSheet_() { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(POSTS_SHEET) || SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; } function getLogSheet_() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName(LOG_SHEET); if (!sheet) { sheet = ss.insertSheet(LOG_SHEET); sheet.appendRow(LOG_HEADERS); sheet.setFrozenRows(1); } return sheet; } function jsonOut_(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } /* -------------------- doGet: 게시글 목록 / 분석 로그 조회 -------------------- */ function doGet(e) { var mode = e.parameter && e.parameter.mode; if (mode === "analytics") { var sheet = getLogSheet_(); var data = sheet.getDataRange().getValues(); var headers = data[0]; var rows = data.slice(1); // days 파라미터로 최근 N일만 필터링 (기본: 전체) var days = e.parameter.days ? parseInt(e.parameter.days, 10) : null; if (days) { var cutoff = new Date(Date.now() - days * 86400000); rows = rows.filter(function(row) { var ts = new Date(row[0]); return ts >= cutoff; }); } // 최대 5000행까지만 반환 (성능 보호) if (rows.length > 5000) rows = rows.slice(rows.length - 5000); var logs = rows.map(function(row) { var obj = {}; headers.forEach(function(h, i) { obj[h] = row[i]; }); return obj; }); return jsonOut_({ result: "success", logs: logs }); } // 기본: 게시글 목록 var sheet = getPostsSheet_(); var data = sheet.getDataRange().getValues(); var headers = data[0]; var rows = data.slice(1).map(function(row) { var obj = {}; headers.forEach(function(h, i) { obj[h] = row[i]; }); return obj; }); return jsonOut_({ result: "success", posts: rows }); } /* -------------------- doPost: 게시글 CRUD + 분석 이벤트 수집 -------------------- */ function doPost(e) { var params = JSON.parse(e.postData.contents); var action = params.action || "create"; // ---- 분석 이벤트 로깅 ---- if (action === "logEvent") { var log = getLogSheet_(); log.appendRow(LOG_HEADERS.map(function(h) { return params[h] != null ? params[h] : ""; })); return jsonOut_({ result: "success" }); } // ---- 게시글 CRUD ---- var sheet = getPostsSheet_(); var data = sheet.getDataRange().getValues(); var headers = data[0]; var idCol = headers.indexOf("id"); function findRowIndexById(id) { for (var i = 1; i < data.length; i++) { if (String(data[i][idCol]) === String(id)) return i + 1; // 시트 행번호(1-base) } return -1; } if (action === "create") { sheet.appendRow([ params.id, params.category, params.emoji, params.title, params.summary, params.tags, params.date, params.content, params.views || 0 ]); } else if (action === "update") { var rowNum = findRowIndexById(params.id); if (rowNum === -1) return jsonOut_({ result: "error", message: "ID를 찾을 수 없습니다." }); var newRow = [ params.id, params.category, params.emoji, params.title, params.summary, params.tags, params.date, params.content, params.views != null ? params.views : data[rowNum - 1][headers.indexOf("views")] ]; sheet.getRange(rowNum, 1, 1, newRow.length).setValues([newRow]); } else if (action === "delete") { var rowNum = findRowIndexById(params.id); if (rowNum === -1) return jsonOut_({ result: "error", message: "ID를 찾을 수 없습니다." }); sheet.deleteRow(rowNum); } else if (action === "incrementView") { var rowNum = findRowIndexById(params.id); if (rowNum !== -1) { var viewsCol = headers.indexOf("views") + 1; var current = Number(sheet.getRange(rowNum, viewsCol).getValue()) || 0; sheet.getRange(rowNum, viewsCol).setValue(current + 1); } } return jsonOut_({ result: "success" }); }

💡 분석로그 시트에는 페이지뷰, 체류시간·이탈, 광고/제휴 링크 클릭, 계산기 사용 이벤트가 모두 한 곳에 쌓입니다. 시트 데이터가 많아지면 정기적으로 오래된 로그를 별도 시트로 옮기거나 삭제하는 것을 권장합니다 (Apps Script 실행시간 제한 고려).
